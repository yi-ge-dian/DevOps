# 服务器执行
```shell
yum -y install nfs-utils
systemctl start nfs-server
systemctl enable nfs-server
systemctl status nfs-server

mkdir -p /nfs/data
cat > /etc/exports << EOF
/nfs/data 10.0.0.0/24(rw,no_root_squash)
EOF

exportfs -rv
showmount -e localhost

```
