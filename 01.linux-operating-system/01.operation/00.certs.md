# 1. CA 机构

1. 生成密钥

```shell
cd /etc/pki/xxx
openssl genpkey -algorithm RSA -pkeyopt rsa_keygen_bits:2048 -out ca.key
```

2. 创建配置文件

```shell
cat > ca.cnf << EOF
[ req ]
defaults_bits = 2048
prompt = no
distinguished_name = xxx
[ xxx ]
C  = CN
ST = Hebei
L  = XiongAn
O  = Zongbu
OU = Operation
CN = Zongbu Root CA
[ v3_ca ]
basicConstraints = CA:TRUE
keyUsage = keyCertSign, cRLSign
EOF
```

3. 生成签名请求文件

```shell
openssl req -new -key ca.key -out ca.csr -config ca.cnf
```

4. 生成 CA 证书

```shell
openssl x509 -req -days 36500 -in ca.csr -signkey ca.key -out ca.crt -extfile ca.cnf -extensions v3_ca
```

5. 查看证书内容

```shell
openssl x509 -text -noout -in ca.crt
```

# 2. 服务器证书

1. 生成服务器私钥 key

```shell
openssl genpkey -algorithm RSA -pkeyopt rsa_keygen_bits:2048 -out server.key
```

2. 创建服务器配置
```shell
cat > server.cnf << EOF
[ req ]
defaults_bits = 2048
prompt = no
distinguished_name = xxx
[ xxx ]
CN = xxx.com
[ v3_ca ]
keyUsage = keyEncipherment, dataEncipherment, digitalSignature
extendedKeyUsage = serverAuth, clientAuth, codeSigning
subjectAltName = @alt_names
[ alt_names ]
DNS.1 = xxx.com
DNS.2 = *.xxx.com
EOF
```

3. 生成签名请求文件
```shell
openssl req -new -key server.key -out server.csr -config server.cnf
```

4. 利用 CA 对服务器证书签名

```shell
openssl x509 -req -days 36500 -in server.csr -CA ca.crt -CAkey ca.key -CAcreateserial -out server.crt -extfile server.cnf -extensions v3_ca
```

# 3. 接入 nginx

```shell
yum -y install nginx
cp server.crt server.key /etc/nginx/cert/
cd /etc/nginx/conf.d
cat > xxx.com << EOF
server {
	listen 80;
	server_name www.xxx.com;
	return 301 https://$server_name$request_uri;
}

server{
	listen 443 ssl;
	server_name www.xxx.com;
	ssl_certificate /etc/nginx/cert/server.crt;
	ssl_certificate_key /etc/nginx/cert/server.key;
	location /{
		root /usr/share/nginx/html;
		index index.html index.htm;
	}
}
nginx -t

curl http://www.xxx.com
curl https://www.xxx.com # 需要配置信任
cp ca.crt /usr/local/share/ca-certificates/
update-ca-certificates
curl https://www.xxx.com 
```
