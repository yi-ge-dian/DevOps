创建数据库
```shell
CREATE DATABASE IF NOT EXISTS 库名
CHARACTER SET charset_name(字符集名称)
COLLATE collation_name(校对规则名称);

CREATE DATABASE IF NOT EXISTS my_database
CHARACTER SET utf8mb4
COLLATE utf8mb4_unicode_ci;
```

创建表
```shell
CREATE TABLE 表名 (字段01 字段01类型 字段01约束或属性信息,字段02 字段02类型 字段02约束或属性信息) 
CHARACTER SET charset_name(字符集名称) 
COLLATE collation_name(校对规则名称);
```


创建用户
```shell
CREATE USER 'my_user'@'192.168.1.%' IDENTIFIED BY '123456';
GRANT ALL ON my_database.* TO 'my_user'@'192.168.1.%';
```
