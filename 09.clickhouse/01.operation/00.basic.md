基本语法

```sql
create database click_db;
show databases;
use click_db;

CREATE TABLE t_web_hits
(
    `timestamp` DateTime,
    `user_id` UInt64,        -- 改为 UInt64（无符号64位整数）
    `url` String,
    `os` String,
    `browser` String,
    `duration_ms` UInt32
)
ENGINE = MergeTree
PARTITION BY toYYYYMM(timestamp)
ORDER BY (timestamp, user_id);

INSERT INTO t_web_hits VALUES
(now(), 1001, '/page/a', 'iOS', 'Safari', 1500),
(now(), 1002, '/page/b', 'Windows', 'Chrome', 3200),
(now(), 1001, '/page/c', 'iOS', 'Safari', 5000);

INSERT INTO t_web_hits VALUES

(now(), 1001, '/page/a', 'iOS', 'Safari', 1500);

-- 手动合并
optimize table t_web_hits;


CREATE TABLE `tb_user`(
    id UInt8,
    name String,
    city String,
    age UInt8
) ENGINE = MergeTree()
PARTITION BY city
PRIMARY KEY(id)
ORDER BY (id, age);
```
