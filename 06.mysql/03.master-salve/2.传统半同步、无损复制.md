旧主已经开始运行了一段时间了，然后搭建主从。

1. 从51导入数据，作为生产环境数据。

```shell
mysql -uroot -p123456 <db.sql
```

2. 主51创建做主从的用户

```shell
mysql -uroot -p123456
mysql> create user 'repuser'@'%' identified by '123456';
mysql> grant replication slave on *.* to  'repuser'@'%';
mysql> flush privileges;
```

3. 从52执行 install-mysql-8.0.45-x86.sh。

4. 半同步/无损同步（增强半同步）插件安装（主从都得做，从库除了下面的，记得取消复制相关参数）

```shell
#手动安装
install plugin rpl_semi_sync_master soname 'semisync_master.so';
install plugin rpl_semi_sync_slave soname 'semisync_slave.so';
#写配置文件
####################################semi sync
plugin_dir = /usr/local/mysql/lib/plugin
plugin_load = "rpl_semi_sync_master=semisync_master.so;rpl_semi_sync_slave=semisync_slave.so"
loose_rpl_semi_sync_master_enabled = 1
loose_rpl_semi_sync_slave_enabled = 1
loose_rpl_semi_sync_master_timeout = 5000
#rpl_semi_sync_master_wait_point = AFTER_COMMIT，半同步模式使用这个，一般不用，我们用下面的 AFTER_COMMIT，即无损同步即可
rpl_semi_sync_master_wait_point = AFTER_SYNC
rpl_semi_sync_master_wait_for_slave_count = 1

systemctl restart mysqld3306
#如果生产环境不能重启数据库，配置好后，通过全局变修改
```

5. 主库备份恢复到从库

```shell
cd /data/3306/backup
# 主库
mysqldump -uroot -proot --single-transaction --master-data=2 --flush-logs --flush-privileges --events --routines --all-databases > backup_fullbackup.sql

# 部分库
mysqldump -uroot -proot --single-transaction --master-data=2 --flush-logs --flush-privileges --events --routines --databases xxx1db xxx2db > backup_xxx1db_xxx2db.sql

# 从库
mysql -uroot -proot < backup_fullbackup.sql

# 主库
# 如果使用 xtrabackup
xtrabackup --defaults-file=/data/3306/etc/my.cnf --uesr=root --password=123456 --backup --stream=tar --parallel=2 --target-dir=/data/3306/backup/fullback | gzip > /data/3306/backup/fullback.tar.gz
# 从库
cd /data/3306/backup
mkdir -p /data/3306/backup/fullbackup
tar zxvf /data/3306/backup/fullbackup.tar.gz -C /data/3306/backup/fullbackup

xtrabackup --defaults-file=/data/3306/etc/my.cnf --prepare --target-dir=/data/3306/backup/fullbackup --parallel=2

systemctl stop mysqld3306
cd /data/3306/data
mv data data_bak
mkdir data
chown -R mysql.mysql data

rsync -avrP /data/3306/backup/fullbackup/* --exclude='xtrabackup_*' /data/3306/data/
chown -R mysql.mysql /data/3306/data
mysql -uroot -p123456
show databases;
```

6. 从库配置

```shell
#记下logfile和pos
head -n 20 backup_fullbackup.sql

#从库建立状态，这里的logfile和pos要从备份的sql文件
CHANGE REPLICATION SOURCE TO
  SOURCE_HOST='10.0.0.51',
  SOURCE_USER='repuser',
  SOURCE_PASSWORD='123456',
  SOURCE_LOG_FILE='mysql-binlog.000001',
  SOURCE_LOG_POS=1198,
  GET_SOURCE_PUBLIC_KEY=1;
  
# 如果用 gtid
cat /data/3306/backup/fullback/xtrabackup.info

show slave status \G;
stop slave;
reset slave;

show master status \G;
reset master;

SET @MYSQLDUMP_TEMP_LOG_BIN = @@SESSION.SQL_LOG_BIN;
SET @@SESSION.SQL_LOG_BIN= 0;
# 从xtrabackup.info拿到
SET @@GLOBAL.GTID_PURGED='27270078-0cd4-11f1-9b0c-0800274c0fb9:1-3';
SET @@SESSION.SQL_LOG_BIN = @MYSQLDUMP_TEMP_LOG_BIN;
CHANGE REPLICATION SOURCE TO
  SOURCE_HOST='10.0.0.51',
  SOURCE_USER='repuser',
  SOURCE_PASSWORD='123456',
  SOURCE_AUTO_POSITION=1,
  GET_SOURCE_PUBLIC_KEY=1;
```
